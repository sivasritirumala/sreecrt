def addNode(v):
    global n
    if v in nodes:
        print(v,"is alredy exist")
    else:
        nodes.append(v)
        n=n+1
    for x in graph:
        x.append(0)
    temp=[]
    for i in range(n):
        temp.append(0)
        graph.append(temp)
def show():
    for i in range(n):
        for j in range(n):
            print(graph[i][j],end="")
        print()
def addEdge(node1,node2):
    if node1 not in nodes:
        print(node1,"not available")
    if node2 not in nodes:
        print(node2,"not available")
    else:
        i=nodes.index(node1)
        j=nodes.index(node2)
        graph[i][j]=1
